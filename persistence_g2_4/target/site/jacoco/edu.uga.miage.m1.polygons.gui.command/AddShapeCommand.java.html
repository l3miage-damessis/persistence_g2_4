package edu.uga.miage.m1.polygons.gui.command;

import edu.uga.miage.m1.polygons.gui.JDrawingPanel;
import edu.uga.miage.m1.polygons.gui.shapes.ShapeType;

public class AddShapeCommand implements Command {

    private JDrawingPanel jDrawingPanel;
    private ShapeType shapeType;
    private int x;
    private int y;

    public AddShapeCommand(JDrawingPanel jDrawingPanel,ShapeType shapeType,int x, int y) {
        this.jDrawingPanel = jDrawingPanel;
        this.shapeType=shapeType;
        this.x=x;
        this.y=y;
    }

    @Override
    public void execute() {
        jDrawingPanel.addShape(shapeType,x,y);
    }

    @Override
    public void undo() {
        jDrawingPanel.removeLastShape();
    }

    @Override
    public void redo() {
        jDrawingPanel.reAddShape();
    }
}
